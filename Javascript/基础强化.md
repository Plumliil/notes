### 循环及类型转换
#### for循环
~~~javascript
for(let i=0;i<10;i++){
    console.log(i)
}
~~~
循环步骤
- 1.声明变量 `let i= 0` 只执行一次
- 2.`if(i<10>){console.log(i)}`
- 3.i++ 可以提到循环体内
- 4.`if(i<10>){console.log(i)}`

for循环可以转化为while循环
~~~javascript
let i=0;
for(;i<10;){
    console.log(i);
    i++
}
while(i<10){
    console.log(i);
    i++
}
~~~
终止循环
~~~javascript
let i=1;
for(;i;){
    console.log(i);
    i++;
    if(i==11){
        i=0
    }
}
~~~
简单斐波那契数列
~~~javascript
// if (n === 1 || n === 2) return 1;
let n = 5;
let n1 = 1;
let n2 = 1;
for (let i = 0; i < n; i++) {
  let res = n1 + n2;
  n1 = n2;
  n2 = res;
  console.log(res)
}
~~~

break 终止循环
return 跳出循环
continue 跳出当前循环,继续剩余步骤

### 类型转化

#### 显式类型转换
**数字类型转换**
非数:
- undefined,NaN转化为NaN
- false,"",null转化为0
数字字符串:'1'=> 1,'2'=> 2,...
- Number():转化为数字
- parseInt():转化为整数,'3.14'=>3
  - 第二个参数表示进制,可以将部分字母转化为数字
  - 16进制 0123456789abcdef
  - parseInt('b',16)=>11
  - parseInt转换,数字开头转数字,非数字开头转NaN,'123abc'=>123,'abc123'=>123
- parseFloat():转化为浮点数
  - '3.14'=>3.14
**布尔类型转换**
- Boolean():转化为布尔类型
  - undefined,null,NaN,"",0,false转化为false,其他都转化为true
**字符串类型转换**
- String()
- xxx.toString 10.toString=>报错,10..toString()=>10,10.1.toString()=>10.1

#### 隐式类型转换
- +'123'=> 123,-'123'=>123 
- *,/,-,% 做运算时都要做隐式转换,string转化为number类型
- >,<,>=,<= 做运算时也会进行隐式转换,首先转换number,其次会转化为ascii值
- == 做运算时也会进行隐式转换
- === 做运算时不进行隐式转换,会进行数据类型检查
- NaN 不等于任何东西
- undefined == 0 false
- null == 0 false
- undefined == null true
~~~javascript
let a='123';
a++;
console.log(a);//124
~~~
~~~javascript
let a = 'a' + 1;
let b = '2' * 2;
console.log(a); // a1
console.log(b); // 4
~~~
~~~javascript
let a = 1 > '2';
let b = '2' > 1;
console.log(a); // false 
console.log(b); // true 
~~~
~~~javascript
let a = 2 > 1 > 3;
let b = 2 > 1 == 1;
let c = 3 > 2 == 1;
let d = 3 > 2 == 2;
console.log(a); // false 2>1=>true=>1<3=>false 
console.log(b); // true 2>1=>true=>1==1=>true
console.log(c); // false 2>1=>true=>1!==1=>true
console.log(d); // false 2>1=>true=>1!==2=>false
~~~
#### isNaN() 
判断是不是NaN
~~~javascript
console.log(isNaN('123')); // false
console.log(isNaN('a')); // true
console.log(isNaN(null)); // false
console.log(isNaN(undefined)); // true
console.log(isNaN("")); // false
~~~
